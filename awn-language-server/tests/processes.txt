FUNCTIONS: exists: MSG -> Bool
           bool2msg: Bool -> MSG
           binary: DATA x MSG -> DATA

CONSTANTS: bool: Bool
           data: DATA
           ip: IP
           msg: MSG
           powip: Pow(IP)

VARIABLES: var: DATA
           msgVar: MSG

proc p1(msgVar, var) :=
        [ bool ]
        [[ var := data ]]
        send(msgVar). p1
        +
        deliver(data). p1
        +
        send(bool2msg(bool)). p1
        +
        deliver(binary(var, msgVar)). p1
